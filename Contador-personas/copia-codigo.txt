import cv2
from ultralytics import YOLO

def contar_personas(video_path=0):
    # Cargar el modelo YOLOv8 pre-entrenado
    model = YOLO("yolov8n.pt")  # puedes usar yolov8s.pt si quieres m치s precisi칩n
    
    # Captura de video (0 = webcam, o pasa un archivo .mp4)
    cap = cv2.VideoCapture(video_path)

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        # Realizar predicci칩n con YOLO
        results = model(frame)

        # Filtrar solo las detecciones de personas (class_id = 0 en COCO dataset)
        personas = [box for box in results[0].boxes if int(box.cls[0]) == 0]

        # Dibujar detecciones en el frame
        for box in personas:
            x1, y1, x2, y2 = box.xyxy[0]
            conf = box.conf[0]
            cv2.rectangle(frame, (int(x1), int(y1)), (int(x2), int(y2)), (0, 255, 0), 2)
            cv2.putText(frame, f"Persona {conf:.2f}", (int(x1), int(y1) - 10),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)

        # Mostrar el conteo en pantalla
        total = len(personas)
        cv2.putText(frame, f"Total Personas: {total}", (20, 50),
                    cv2.FONT_HERSHEY_SIMPLEX, 1.2, (0, 0, 255), 3)

        # Mostrar el frame
        cv2.imshow("Conteo de Personas - YOLOv8", frame)

        # Salir con la tecla 'q'
        if cv2.waitKey(1) & 0xFF == ord("q"):
            break

    cap.release()
    cv2.destroyAllWindows()

if __name__ == "__main__":
    # Usa la webcam por defecto
    contar_personas(0)
